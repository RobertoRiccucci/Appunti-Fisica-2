\begin{tikzpicture}[
  >={Latex[length=3mm,width=2.2mm]},
  every node/.style={font=\small},
  axis/.style={->,thick,black},
  evec/.style={->,line width=1.3pt,ered},
  edge/.style={->,line width=0.7pt,black}
]

% Assi coordinati (prospettiva)
\coordinate (O)  at (0,0);
\coordinate (X)  at (0,3.2);     % x verticale
\coordinate (Y)  at (-2.5,-1.6); % y verso il lettore
\coordinate (Z)  at ( 4.2,0.8);  % z a destra/alto

\draw[axis] (O) -- (X) node[above] {$x$};
\draw[axis] (O) -- (Y) node[below left] {$y$};
\draw[axis] (O) -- (Z) node[right] {$z$};

% Parallelogramma ABCD (superficie)
\coordinate (A) at ( 0.4,-0.4);
\coordinate (D) at ( 3.0, 0.1);
\coordinate (C) at ( 2.4,-1.6);
\coordinate (B) at (-0.2,-2.1);

\fill[surfblue,opacity=0.9] (A) -- (D) -- (C) -- (B) -- cycle;
\draw[rounded corners=1pt,thick,black] (A) -- (D) -- (C) -- (B) -- cycle;

% Etichette vertici
\node[above]       at (A) {$A$};
\node[above right] at (D) {$D$};
\node[below right] at (C) {$C$};
\node[below left]  at (B) {$B$};

% Lati differenziali: dz (BC) e dy (DC)
\draw[edge] ($(B)!0.15!(C)$) -- ($(B)!0.85!(C)$);
\node[below] at ($(B)!0.5!(C)$) {$dz$};
\draw[edge] ($(D)!0.15!(C)$) -- ($(D)!0.85!(C)$);
\node[right] at ($(D)!0.5!(C)$) {$dy$};

% Vettore E al centro della superficie
\coordinate (M) at ($(A)!0.5!(C)$);
\draw[evec] (M) -- ++(0.5,1.3) node[above left] {$\mathbf{E}$};

% Vettore E al vertice D (uscente)
\draw[evec] (D) -- ++(0.6,1.6) node[above] {$\mathbf{E}$};

% Cavatappi (simbolo orientazione) sotto
\begin{scope}[shift={(1.1,-3.6)}]
  \draw[thick] (-0.25,0) -- (0.25,0) -- (0.15,-1.2) -- (-0.15,-1.2) -- cycle;
  \foreach \y in {-0.1,-0.3,-0.5,-0.7,-0.9,-1.1}{%
     \draw[thin] (-0.22,\y) -- (0.22,\y+0.08);
  }
  \draw[->,thick] (0,0.1) -- (0,0.7);
  \draw[->,thick] (0.35,-1.0) arc (0:-330:0.35);
\end{scope}

\end{tikzpicture}
